# Database

```psql
CREATE DATABASE llm_chatbot_db;
CREATE ROLE llm_chatbot_user WITH LOGIN PASSWORD 'llm_chatbot_password';
GRANT ALL PRIVILEGES ON DATABASE llm_chatbot_db TO llm_chatbot_user;
GRANT ALL PRIVILEGES ON ALL TABLES IN SCHEMA public TO llm_chatbot_user;
GRANT ALL PRIVILEGES ON ALL SEQUENCES IN SCHEMA public TO llm_chatbot_user;
GRANT ALL PRIVILEGES ON ALL FUNCTIONS IN SCHEMA public TO llm_chatbot_user;
ALTER DEFAULT PRIVILEGES IN SCHEMA public GRANT ALL PRIVILEGES ON TABLES TO llm_chatbot_user;
ALTER DEFAULT PRIVILEGES IN SCHEMA public GRANT ALL PRIVILEGES ON SEQUENCES TO llm_chatbot_user;
ALTER DEFAULT PRIVILEGES IN SCHEMA public GRANT ALL PRIVILEGES ON FUNCTIONS TO llm_chatbot_user;
```
